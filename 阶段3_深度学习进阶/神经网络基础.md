## 神经网络概述

### 什么是神经网络
神经网络是一种模仿生物神经网络的计算模型，由大量相互连接的节点（神经元）组成，能够学习复杂的非线性关系。

### 发展历史
1. **感知机(1957)**：Frank Rosenblatt提出
2. **反向传播(1986)**：Rumelhart、Hinton、Williams提出
3. **深度学习复兴(2006)**：Hinton提出深度信念网络
4. **现代深度学习(2012+)**：AlexNet在ImageNet上取得突破

## 第一周：感知机与多层感知机

### 1.1 感知机
**数学模型：**
```
y = sign(w·x + b)
其中 w 是权重，x 是输入，b 是偏置
```

**Python实现：**
```python
import numpy as np

class Perceptron:
    def __init__(self, n_inputs, learning_rate=0.01):
        self.weights = np.random.randn(n_inputs)
        self.bias = 0
        self.lr = learning_rate
    
    def activation(self, z):
        """阶跃函数"""
        return 1 if z >= 0 else 0
    
    def predict(self, x):
        z = np.dot(self.weights, x) + self.bias
        return self.activation(z)
    
    def train(self, X, y, epochs=100):
        for _ in range(epochs):
            for xi, yi in zip(X, y):
                y_pred = self.predict(xi)
                error = yi - y_pred
                self.weights += self.lr * error * xi
                self.bias += self.lr * error

# 测试感知机（与门）
X = np.array([[0, 0], [0, 1], [1, 0], [1, 1]])
y = np.array([0, 0, 0, 1])

p = Perceptron(2)
p.train(X, y)

for xi in X:
    print(f"输入: {xi}, 输出: {p.predict(xi)}")
```

### 1.2 多层感知机(MLP)
**结构：**
- 输入层：接收原始数据
- 隐藏层：一个或多个全连接层
- 输出层：产生最终预测

**Python实现：**
```python
class MLP:
    def __init__(self, layer_sizes, learning_rate=0.01):
        self.layer_sizes = layer_sizes
        self.lr = learning_rate
        self.weights = []
        self.biases = []
        
        # 初始化权重和偏置
        for i in range(len(layer_sizes) - 1):
            w = np.random.randn(layer_sizes[i], layer_sizes[i+1]) * 0.01
            b = np.zeros(layer_sizes[i+1])
            self.weights.append(w)
            self.biases.append(b)
    
    def sigmoid(self, z):
        return 1 / (1 + np.exp(-z))
    
    def sigmoid_derivative(self, z):
        return z * (1 - z)
    
    def forward(self, x):
        self.activations = [x]
        self.z_values = []
        
        for i, (w, b) in enumerate(zip(self.weights, self.biases)):
            z = np.dot(self.activations[-1], w) + b
            self.z_values.append(z)
            a = self.sigmoid(z)
            self.activations.append(a)
        
        return self.activations[-1]
    
    def backward(self, x, y):
        m = x.shape[0]
        delta = self.activations[-1] - y
        
        for i in range(len(self.weights) - 1, -1, -1):
            dw = np.dot(self.activations[i].T, delta) / m
            db = np.sum(delta, axis=0) / m
            
            if i > 0:
                delta = np.dot(delta, self.weights[i].T) * self.sigmoid_derivative(self.activations[i])
            
            self.weights[i] -= self.lr * dw
            self.biases[i] -= self.lr * db
    
    def train(self, X, y, epochs=1000):
        for epoch in range(epochs):
            output = self.forward(X)
            self.backward(X, y)
            
            if epoch % 100 == 0:
                loss = np.mean((y - output) ** 2)
                print(f"Epoch {epoch}, Loss: {loss:.4f}")

# 测试MLP（XOR问题）
X = np.array([[0, 0], [0, 1], [1, 0], [1, 1]])
y = np.array([[0], [1], [1], [0]])

mlp = MLP([2, 4, 1], learning_rate=0.1)
mlp.train(X, y, epochs=5000)

print("\n预测结果:")
for xi, yi in zip(X, y):
    pred = mlp.forward(xi.reshape(1, -1))
    print(f"输入: {xi}, 真实: {yi[0]}, 预测: {pred[0][0]:.4f}")
```

### 2.2 损失函数
**回归损失函数：**
```python
def mse_loss(y_true, y_pred):
    """均方误差"""
    return np.mean((y_true - y_pred) ** 2)

def mae_loss(y_true, y_pred):
    """平均绝对误差"""
    return np.mean(np.abs(y_true - y_pred))

def huber_loss(y_true, y_pred, delta=1.0):
    """Huber损失（对异常值鲁棒）"""
    residual = np.abs(y_true - y_pred)
    condition = residual < delta
    loss = np.where(condition, 0.5 * residual**2, delta * residual - 0.5 * delta**2)
    return np.mean(loss)
```

**分类损失函数：**
```python
def binary_crossentropy(y_true, y_pred):
    """二元交叉熵"""
    epsilon = 1e-15
    y_pred = np.clip(y_pred, epsilon, 1 - epsilon)
    return -np.mean(y_true * np.log(y_pred) + (1 - y_true) * np.log(1 - y_pred))

def categorical_crossentropy(y_true, y_pred):
    """分类交叉熵"""
    epsilon = 1e-15
    y_pred = np.clip(y_pred, epsilon, 1 - epsilon)
    return -np.mean(np.sum(y_true * np.log(y_pred), axis=1))
```

## 第三周：优化算法

### 3.1 梯度下降变体
```python
class Optimizer:
    def __init__(self, learning_rate=0.01):
        self.lr = learning_rate
    
    def sgd(self, gradients):
        """随机梯度下降"""
        return -self.lr * gradients
    
    def sgd_momentum(self, gradients, velocity, beta=0.9):
        """带动量的SGD"""
        velocity = beta * velocity + gradients
        return -self.lr * velocity, velocity
    
    def rmsprop(self, gradients, cache, beta=0.9, epsilon=1e-8):
        """RMSprop"""
        cache = beta * cache + (1 - beta) * gradients**2
        return -self.lr * gradients / (np.sqrt(cache) + epsilon), cache
    
    def adam(self, gradients, m, v, t, beta1=0.9, beta2=0.999, epsilon=1e-8):
        """Adam优化器"""
        m = beta1 * m + (1 - beta1) * gradients
        v = beta2 * v + (1 - beta2) * gradients**2
        
        m_hat = m / (1 - beta1**t)
        v_hat = v / (1 - beta2**t)
        
        update = -self.lr * m_hat / (np.sqrt(v_hat) + epsilon)
        return update, m, v
```

### 3.2 学习率调度
```python
class LRScheduler:
    def __init__(self, initial_lr):
        self.initial_lr = initial_lr
    
    def step_decay(self, epoch, drop_rate=0.5, epochs_per_drop=10):
        """阶梯下降"""
        return self.initial_lr * (drop_rate ** (epoch // epochs_per_drop))
    
    def exponential_decay(self, epoch, decay_rate=0.95):
        """指数下降"""
        return self.initial_lr * (decay_rate ** epoch)
    
    def cosine_annealing(self, epoch, total_epochs, eta_min=0):
        """余弦退火"""
        return eta_min + 0.5 * (self.initial_lr - eta_min) * (1 + np.cos(np.pi * epoch / total_epochs))
    
    def warmup_cosine(self, epoch, warmup_epochs, total_epochs):
        """带预热的余弦退火"""
        if epoch < warmup_epochs:
            return self.initial_lr * (epoch + 1) / warmup_epochs
        else:
            progress = (epoch - warmup_epochs) / (total_epochs - warmup_epochs)
            return 0.5 * self.initial_lr * (1 + np.cos(np.pi * progress))
```

## 第四周：正则化技术

### 4.1 Dropout
```python
class Dropout:
    def __init__(self, rate=0.5):
        self.rate = rate
    
    def forward(self, x, training=True):
        if training:
            self.mask = np.random.binomial(1, 1 - self.rate, size=x.shape)
            return x * self.mask / (1 - self.rate)
        else:
            return x
    
    def backward(self, grad_output):
        return grad_output * self.mask / (1 - self.rate)
```

### 4.2 Batch Normalization
```python
class BatchNormalization:
    def __init__(self, n_features, momentum=0.9, epsilon=1e-5):
        self.gamma = np.ones(n_features)
        self.beta = np.zeros(n_features)
        self.momentum = momentum
        self.epsilon = epsilon
        
        self.running_mean = np.zeros(n_features)
        self.running_var = np.ones(n_features)
    
    def forward(self, x, training=True):
        if training:
            batch_mean = np.mean(x, axis=0)
            batch_var = np.var(x, axis=0)
            
            self.running_mean = self.momentum * self.running_mean + (1 - self.momentum) * batch_mean
            self.running_var = self.momentum * self.running_var + (1 - self.momentum) * batch_var
            
            x_norm = (x - batch_mean) / np.sqrt(batch_var + self.epsilon)
        else:
            x_norm = (x - self.running_mean) / np.sqrt(self.running_var + self.epsilon)
        
        return self.gamma * x_norm + self.beta
```

### 4.3 权重正则化
```python
def l1_regularization(weights, lambda_):
    """L1正则化"""
    return lambda_ * np.sum(np.abs(weights))

def l2_regularization(weights, lambda_):
    """L2正则化"""
    return lambda_ * 0.5 * np.sum(weights**2)

def elastic_net(weights, lambda_l1, lambda_l2):
    """弹性网络"""
    return l1_regularization(weights, lambda_l1) + l2_regularization(weights, lambda_l2)
```

## PyTorch实现

### 完整神经网络示例
```python
import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader, TensorDataset

class SimpleNN(nn.Module):
    def __init__(self, input_size, hidden_size, output_size):
        super(SimpleNN, self).__init__()
        self.fc1 = nn.Linear(input_size, hidden_size)
        self.relu = nn.ReLU()
        self.dropout = nn.Dropout(0.5)
        self.fc2 = nn.Linear(hidden_size, output_size)
    
    def forward(self, x):
        x = self.fc1(x)
        x = self.relu(x)
        x = self.dropout(x)
        x = self.fc2(x)
        return x

# 创建数据
X = torch.randn(1000, 10)
y = torch.randint(0, 2, (1000,))

# 创建数据加载器
dataset = TensorDataset(X, y)
dataloader = DataLoader(dataset, batch_size=32, shuffle=True)

# 创建模型
model = SimpleNN(10, 64, 2)
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)

# 训练循环
def train_model(model, dataloader, criterion, optimizer, epochs=100):
    model.train()
    
    for epoch in range(epochs):
        total_loss = 0
        
        for batch_X, batch_y in dataloader:
            optimizer.zero_grad()
            outputs = model(batch_X)
            loss = criterion(outputs, batch_y)
            loss.backward()
            optimizer.step()
            
            total_loss += loss.item()
        
        if (epoch + 1) % 10 == 0:
            print(f"Epoch {epoch+1}, Loss: {total_loss/len(dataloader):.4f}")

# 训练模型
train_model(model, dataloader, criterion, optimizer)
```
