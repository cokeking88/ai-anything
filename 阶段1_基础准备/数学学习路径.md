import numpy as np

# 特征值分解
A = np.array([[4, 2], [1, 3]])
eigenvalues, eigenvectors = np.linalg.eig(A)

print(f"特征值：{eigenvalues}")
print(f"特征向量：\n{eigenvectors}")

# 验证：A * v = λ * v
for i in range(len(eigenvalues)):
    left = np.dot(A, eigenvectors[:, i])
    right = eigenvalues[i] * eigenvectors[:, i]
    print(f"验证特征对 {i}: {np.allclose(left, right)}")

# SVD分解
U, S, Vt = np.linalg.svd(A)
print(f"\nSVD分解：")
print(f"U矩阵：\n{U}")
print(f"奇异值：{S}")
print(f"Vt矩阵：\n{Vt}")
```

# 概率计算
# 抛硬币概率
def coin_probability(heads, tails):
    total = heads + tails
    p_heads = heads / total
    p_tails = tails / total
    return p_heads, p_tails

p_h, p_t = coin_probability(5, 5)
print(f"公平硬币概率：正面={p_h:.2f}，反面={p_t:.2f}")

# 贝叶斯定理应用
# 疾病检测问题
def bayes_theorem(p_disease, p_positive_given_disease, p_positive_given_healthy):
    p_positive = (p_positive_given_disease * p_disease + 
                  p_positive_given_healthy * (1 - p_disease))
    p_disease_given_positive = (p_positive_given_disease * p_disease) / p_positive
    return p_disease_given_positive

# 已知条件
p_disease = 0.01  # 疾病患病率1%
p_positive_given_disease = 0.95  # 真阳性率95%
p_positive_given_healthy = 0.05  # 假阳性率5%

result = bayes_theorem(p_disease, p_positive_given_disease, p_positive_given_healthy)
print(f"检测阳性时患病概率：{result:.4f}")
```

import numpy as np

# 数值导数
def numerical_derivative(f, x, h=1e-5):
    """计算函数在x处的数值导数"""
    return (f(x + h) - f(x - h)) / (2 * h)

# 测试函数
def f(x):
    return x**2 + 2*x + 1

# 计算导数
x = 2
derivative = numerical_derivative(f, x)
print(f"函数 f(x)=x²+2x+1 在 x={x} 处的导数：{derivative:.4f}")
print(f"理论导数：2*x+2 = {2*x+2}")

# 多元函数梯度
def gradient_2d(f, x, y, h=1e-5):
    """计算二元函数的梯度"""
    df_dx = (f(x + h, y) - f(x - h, y)) / (2 * h)
    df_dy = (f(x, y + h) - f(x, y - h)) / (2 * h)
    return np.array([df_dx, df_dy])

# 测试函数：f(x,y) = x² + y²
def f_2d(x, y):
    return x**2 + y**2

# 计算梯度
x, y = 1, 2
grad = gradient_2d(f_2d, x, y)
print(f"函数 f(x,y)=x²+y² 在点({x},{y})处的梯度：{grad}")
print(f"理论梯度：(2x, 2y) = ({2*x}, {2*y})")
```

import numpy as np
import matplotlib.pyplot as plt

# 梯度下降法实现
def gradient_descent(f, gradient, start, learning_rate=0.1, n_iterations=100):
    """梯度下降算法"""
    x = start
    history = [x]
    
    for i in range(n_iterations):
        grad = gradient(x)
        x = x - learning_rate * grad
        history.append(x)
    
    return x, history

# 测试函数：f(x) = x²
def f(x):
    return x**2

def gradient_f(x):
    return 2*x

# 运行梯度下降
start = 5
learning_rate = 0.1
n_iterations = 20

minimum, history = gradient_descent(f, gradient_f, start, learning_rate, n_iterations)

print(f"起始点：{start}")
print(f"找到的最小值点：{minimum:.6f}")
print(f"函数值：{f(minimum):.6f}")

# 可视化收敛过程
plt.figure(figsize=(10, 6))
x_vals = np.linspace(-6, 6, 100)
y_vals = f(x_vals)
plt.plot(x_vals, y_vals, 'b-', label='f(x) = x²')
plt.plot(history, [f(x) for x in history], 'ro-', label='梯度下降路径')
plt.title('梯度下降收敛过程')
plt.xlabel('x')
plt.ylabel('f(x)')
plt.legend()
plt.grid(True)

import numpy as np

# 熵计算
def entropy(probabilities):
    """计算熵 H(p) = -∑ p(x) * log2(p(x))"""
    # 移除概率为0的项
    p = probabilities[probabilities > 0]
    return -np.sum(p * np.log2(p))

# 示例：抛硬币
fair_coin = np.array([0.5, 0.5])  # 公平硬币
biased_coin = np.array([0.9, 0.1])  # 偏硬币

H_fair = entropy(fair_coin)
H_biased = entropy(biased_coin)

print(f"公平硬币的熵：{H_fair:.4f}")
print(f"偏硬币的熵：{H_biased:.4f}")

# KL散度
def kl_divergence(p, q):
    """计算KL散度 D_KL(p||q) = ∑ p(x) * log(p(x)/q(x))"""
    # 移除概率为0的项
    mask = (p > 0) & (q > 0)
    p = p[mask]
    q = q[mask]
    return np.sum(p * np.log(p / q))

# 示例
p = np.array([0.5, 0.5])
q = np.array([0.9, 0.1])

kl = kl_divergence(p, q)
print(f"KL散度 D_KL(p||q)：{kl:.4f}")

# 交叉熵
def cross_entropy(p, q):
    """计算交叉熵 H(p,q) = -∑ p(x) * log(q(x))"""
    # 移除概率为0的项
    mask = (p > 0) & (q > 0)
    p = p[mask]
    q = q[mask]
    return -np.sum(p * np.log(q))

ce = cross_entropy(p, q)
print(f"交叉熵 H(p,q)：{ce:.4f}")
```
